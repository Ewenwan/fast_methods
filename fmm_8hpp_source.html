93  virtual double solveEikonal
94  (const int & idx) {
95  unsigned int a = grid_t::getNDims(); // a parameter of the Eikonal equation.
96 
97  double updatedT;
98  sumT = 0;
99  sumTT = 0;
100 
101  for (unsigned int dim = 0; dim < grid_t::getNDims(); ++dim) {
102  double minTInDim = grid_->getMinValueInDim(idx, dim);
103  if (!isinf(minTInDim)) {
104  Tvalues[dim] = minTInDim;
105  sumT += Tvalues[dim];
106  TTvalues[dim] = Tvalues[dim]*Tvalues[dim];
107  sumTT += TTvalues[dim];
108  }
109  else {
110  Tvalues[dim] = 0;
111  TTvalues[dim] = 0;
112  a -=1 ;
113  }
114  }
115 
116  double b = -2*sumT;
117  double c = sumTT - grid_->getLeafSize() * grid_->getLeafSize()/(grid_->getCell(idx).getVelocity()*grid_->getCell(idx).getVelocity());
118  double quad_term = b*b - 4*a*c;
119  if (quad_term < 0) {
120  double minT = *(std::min_element(Tvalues.begin(), Tvalues.end()));
121  updatedT = grid_->getLeafSize() * grid_->getLeafSize()/(grid_->getCell(idx).getVelocity()*grid_->getCell(idx).getVelocity()) + minT;
122  }
123  else
124  updatedT = (-b + sqrt(quad_term))/(2*a);
125 
126  if (heurStrategy_ == TIME)
127  grid_->getCell(idx).setHeuristicTime( getPrecomputedDistance(idx)/grid_->getCell(idx).getVelocity() );
128  else if (heurStrategy_ == DISTANCE)
129  grid_->getCell(idx).setHeuristicTime( getPrecomputedDistance(idx) );
130 
131  return updatedT;
132  }

135  virtual void computeInternal
136  () {
137  if (!setup_)
138  setup();
139 
140  unsigned int j = 0;
141  unsigned int n_neighs = 0;
142  bool stopWavePropagation = false;
143 
144  // Algorithm initialization
145  for (unsigned int &i: init_points_) { // For each initial point
146  grid_->getCell(i).setArrivalTime(0);
147  narrow_band_.push( &(grid_->getCell(i)) );
148  }
149 
150  // Main loop.
151  while (!stopWavePropagation && !narrow_band_.empty()) {
152  unsigned int idxMin = narrow_band_.popMinIdx();
153  n_neighs = grid_->getNeighbors(idxMin, neighbors);
154  grid_->getCell(idxMin).setState(FMState::FROZEN);
155 
156  for (unsigned int s = 0; s < n_neighs; ++s) {
157  j = neighbors[s];
158  // If Frozen or obstacle
159  if ((grid_->getCell(j).getState() == FMState::FROZEN) || grid_->getCell(j).isOccupied())
160  continue;
161  else {
162  double new_arrival_time = solveEikonal(j);
163  // Updating narrow band if necessary.
164  if (grid_->getCell(j).getState() == FMState::NARROW) {
165  if (new_arrival_time < grid_->getCell(j).getArrivalTime()) {
166  grid_->getCell(j).setArrivalTime(new_arrival_time);
167  narrow_band_.increase( &(grid_->getCell(j)) );
168  }
169  }
170  else {
171  grid_->getCell(j).setState(FMState::NARROW);
172  grid_->getCell(j).setArrivalTime(new_arrival_time);
173  narrow_band_.push( &(grid_->getCell(j)) );
174  } // neighbors open.
175  } // neighbors not frozen.
176  if (idxMin == goal_idx_)
177  stopWavePropagation = true;
178  } // For each neighbor.
179  } // while narrow band not empty
180  }
